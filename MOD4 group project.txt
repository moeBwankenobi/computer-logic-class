START
    INPUT grossPay
    CALCULATE stateTax = grossPay * 5.6
    CALCULATE federalTax = grossPay * 7.9
    CALCULATE Totaldeductions = federalTax + stateTax
    CALCULATE netPay = grossPay - Totaldeductions
    OUTPUT stateTax
    OUTPUT federalTax
    OUTPUT Totaldeductions
    OUTPUT netPay
END
