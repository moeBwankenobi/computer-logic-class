START
    Declarations
        NUM grossPay
        NUM stateTax
        NUM federalTax
        NUM Totaldeductions
    INPUT grossPay
    stateTax = grossPay * 5.6
    federalTax = grossPay * 7.9
    Totaldeductions = federalTax + stateTax
    netPay = grossPay - Totaldeductions
    OUTPUT stateTax
    OUTPUT federalTax
    OUTPUT Totaldeductions
    OUTPUT netPay
END
